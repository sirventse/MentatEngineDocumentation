139 void SetIterators(ME::Iterator<T>* it, ME::Iterator<T>* end, T** first) {
140 ME::ContainerFacade<T> container{ ME::ECS::GetComponentList<T>() };
141
142 *first = nullptr;
143
144 if (container.isEmpty()) {
145 // deja it y end en un estado consistente

147 *end = container.end();
148 it->SetEmpty(true);
149 end->SetEmpty(true);

152
153 *it = container.begin();
154 *end = container.end();
155 it->SetEmpty(false);
156 end->SetEmpty(false);

159 *first = **it;
160 }

162 it->SetEmpty(true);
