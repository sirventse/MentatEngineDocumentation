19static Vec3 NormalizeSafe(const Vec3& v) {
20 float len = sqrtf(v.x_ * v.x_ + v.y_ * v.y_ + v.z_ * v.z_);
21 if (len < 1e-6f) return Vec3{ 0.0f, 1.0f, 0.0f };
22 return Vec3{ v.x_ / len, v.y_ / len, v.z_ / len };
23}

25static Vec3 Cross(const Vec3& a, const Vec3& b) {
26 return Vec3{
27 a.y_ * b.z_ - a.z_ * b.y_,
28 a.z_ * b.x_ - a.x_ * b.z_,
29 a.x_ * b.y_ - a.y_ * b.x_
30 };
31}

33static Vec3 Sub(const Vec3& a, const Vec3& b) {
34 return Vec3{ a.x_ - b.x_, a.y_ - b.y_, a.z_ - b.z_ };
35}

37static void ComputeSmoothNormals(
38 const std::vector<Vec3>& positions,
39 const std::vector<unsigned int>& indices,
40 std::vector<Vec3>& outNormals
41) {
42 outNormals.assign(positions.size(), Vec3{ 0,0,0 });
43
44 for (size_t i = 0; i + 2 < indices.size(); i += 3) {
45 unsigned int i0 = indices[i];
46 unsigned int i1 = indices[i + 1];
47 unsigned int i2 = indices[i + 2];
48
49 const auto& p0 = positions[i0];
50 const auto& p1 = positions[i1];
51 const auto& p2 = positions[i2];
52
53 Vec3 e1 = Sub(p1, p0);
54 Vec3 e2 = Sub(p2, p0);
55 Vec3 n = Cross(e1, e2); // sin normalizar para ponderar por área
56
57 outNormals[i0].x_ += n.x_; outNormals[i0].y_ += n.y_; outNormals[i0].z_ += n.z_;
58 outNormals[i1].x_ += n.x_; outNormals[i1].y_ += n.y_; outNormals[i1].z_ += n.z_;
59 outNormals[i2].x_ += n.x_; outNormals[i2].y_ += n.y_; outNormals[i2].z_ += n.z_;
60 }
61
62 for (auto& n : outNormals) n = NormalizeSafe(n);
63}
